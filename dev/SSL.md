## 証明書ファイルの内容を確認
openssl x509 -text -noout -in /[FilePath]/[CertFile]

## 秘密鍵ファイルの内容を確認
openssl rsa -text -noout -in /[FilePath]/[KeyFile]

## CSRファイルの内容を確認
penssl req -text -noout -in /[FilePath]/[CSRFile]
